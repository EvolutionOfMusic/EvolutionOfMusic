Example

C Side

		#include  <stdio.h>
		#include  <stdlib.h>
		
		int main(){
		  char buffer[100];
		
		  sprintf(buffer, "python3 hello.py");
		  puts("starting python3 in C"); 
		  system(buffer); // Runs command in bash; Stops C from executing
		  puts("resuming C");
		  return 0;
		}
